  <script>
    let currentObject = 'kotak';
    let currentSurface = 'smooth';
    let isDragging = false;
    let startX, startY;
    let objX = 100, objY = 120;

    const obj = document.getElementById('moving-object');
    const arrow = document.getElementById('force-arrow');
    const playground = document.getElementById('playground');
    const infoText = document.getElementById('info-text');

    function selectObject(type) {
      currentObject = type;
      document.querySelectorAll('.btn-choice').forEach(btn => btn.classList.remove('active'));
      event.target.classList.add('active');
      
      // Ganti ikon
      if (type === 'kotak') obj.innerHTML = 'üì¶';
      else if (type === 'bola') obj.innerHTML = '‚öΩ';
      else obj.innerHTML = 'üéà';
      
      resetObject();
    }

    function selectSurface(type) {
      currentSurface = type;
      document.querySelectorAll('.btn-choice').forEach(btn => btn.classList.remove('active'));
      event.target.classList.add('active');
      
      playground.className = `surface-${type}`;
      resetObject();
    }

    function resetObject() {
      objX = 100;
      objY = 120;
      obj.style.left = objX + 'px';
      obj.style.top = objY + 'px';
      arrow.style.display = 'none';
      infoText.textContent = "Sentuh dan tahan benda, lalu geser untuk menarik!";
    }

    // Touch/Drag Logic
    obj.addEventListener('touchstart', handleStart, { passive: false });
    obj.addEventListener('touchmove', handleMove, { passive: false });
    obj.addEventListener('touchend', handleEnd);

    obj.addEventListener('mousedown', (e) => {
      isDragging = true;
      startX = e.clientX;
      startY = e.clientY;
      e.preventDefault();
    });

    window.addEventListener('mousemove', (e) => {
      if (!isDragging) return;
      applyForce(e.clientX, e.clientY);
    });

    window.addEventListener('mouseup', () => {
      isDragging = false;
      releaseObject();
    });

    function handleStart(e) {
      isDragging = true;
      const touch = e.touches[0];
      startX = touch.clientX;
      startY = touch.clientY;
      e.preventDefault();
    }

    function handleMove(e) {
      if (!isDragging) return;
      const touch = e.touches[0];
      applyForce(touch.clientX, touch.clientY);
      e.preventDefault();
    }

    function handleEnd() {
      isDragging = false;
      releaseObject();
    }

    function applyForce(endX, endY) {
      const dx = endX - startX;
      const dy = endY - startY;
      
      // Tampilkan panah gaya
      arrow.style.left = (objX + 30) + 'px';
      arrow.style.top = (objY + 30) + 'px';
      arrow.style.display = 'block';
      
      // Arah panah sederhana
      if (Math.abs(dx) > Math.abs(dy)) {
        arrow.innerHTML = dx > 0 ? '‚Üí' : '‚Üê';
      } else {
        arrow.innerHTML = dy > 0 ? '‚Üì' : '‚Üë';
      }

      // Gerakkan benda (simulasi sederhana)
      let moveX = objX + dx * 0.3;
      let moveY = objY + dy * 0.3;

      // Batas playground
      moveX = Math.max(0, Math.min(playground.offsetWidth - 60, moveX));
      moveY = Math.max(0, Math.min(playground.offsetHeight - 60, moveY));

      objX = moveX;
      objY = moveY;
      obj.style.left = objX + 'px';
      obj.style.top = objY + 'px';

      // Info dinamis
      const friction = currentSurface === 'rough' ? ' besar' : ' kecil';
      const mass = currentObject === 'bola' ? 'berat' : (currentObject === 'balon' ? 'ringan' : 'sedang');
      infoText.innerHTML = `Gaya diberikan! Permukaan <strong>${currentSurface === 'rough' ? 'kasar' : 'licin'}</strong> ‚Üí gesekan${friction}.<br>Benda ${mass} ‚Üí percepatan ${mass === 'ringan' ? 'besar' : 'sedang'}.`;
    }

    function releaseObject() {
      arrow.style.display = 'none';
      setTimeout(() => {
        infoText.textContent = "Coba tarik lagi! Arah dan kekuatan gaya memengaruhi gerak.";
      }, 1000);
    }

    // Kuis
    function checkQuiz(answer) {
      const fb = document.getElementById('quiz-feedback');
      if (answer === 'kanan') {
        fb.innerHTML = '‚úÖ Benar! Gaya dan arah gerak searah.';
        fb.className = 'mt-2 font-medium text-green-700';
      } else {
        fb.innerHTML = '‚ùå Salah. Gaya ke kanan ‚Üí benda bergerak ke kanan.';
        fb.className = 'mt-2 font-medium text-red-700';
      }
      fb.classList.remove('hidden');
    }
